171 /** Print a list (one per line) of shared_ptr which refer to the same object
172  *
173  * @param ptr Use the object pointed to by this shared_ptr
174  * @param self include or omit a line for this shared_ptr
175  * @param weak include a line for each weak_ptr (not implemented)

179  @endcode

182 inline ::detail::ref_shower<T> show_referrers(const std::tr1::shared_ptr<T>& ptr, bool self=true, bool weak=false)

205 /** A semi-hack to help with migration from std::auto_ptr to std::unique_ptr,
206  * and avoid copious deprecation warning spam
207  * which may be hiding legitimate issues.
208  *
209  * Provides epics::auto_ptr<T> and epics::swap()
210  *
211  * epics::auto_ptr<T> is std::auto_ptr<T> for c++98
212  * and std::unique_ptr<T> for >= c++11.
213  *
214  * epics::swap() is the only supported operation.
215  * copy/assignment/return are not supported
216  * (use auto_ptr or unique_ptr explicitly).
217  */
218 #if __cplusplus>=201103L
219 template<typename T>
220 using auto_ptr = std::unique_ptr<T>;
221 template<typename T>
222 static inline void swap(auto_ptr<T>& lhs, auto_ptr<T>& rhs) {
223  lhs.swap(rhs);
224 }
225 #else
226 using std::auto_ptr;
227 template<typename T>
228 static inline void swap(auto_ptr<T>& lhs, auto_ptr<T>& rhs) {
229  auto_ptr<T> temp(lhs);
230  lhs = rhs;
231  rhs = temp;
232 }
233 #endif
234 }
